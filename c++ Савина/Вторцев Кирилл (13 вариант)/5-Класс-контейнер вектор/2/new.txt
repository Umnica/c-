/*Перед каждым k-ым по счету элементом вставить 0*/
#include <iostream>
#include <vector>
using namespace std;
int main() {
    // Создаем начальный вектор с n элементами 
    size_t n;
    cout << "n= ";
    cin >> n;
    vector <int> iVec;

    // Ввод элементов вектора iVec ч/з push_back()
    for (size_t i = 0; i < n; i++)
        iVec.push_back(i+1);

    for (size_t i = 0; i < iVec.size(); i++)
        cout << iVec[i] << ' ';
    cout << '\n';

    // Ввод k
    size_t k;
    cout << "k= ";
    cin >> k;
    k++;

    // Вставляем нули перед каждым k-ым эл-ом
    for (size_t i = k - 2; i < iVec.size(); i += k)
        iVec.insert(iVec.begin() + i, 0);

    // Выводим полученный вектор
    for (size_t i = 0; i < iVec.size(); i++)
        cout << iVec[i] << ' ';
    cout << '\n';
    return 0;
}
