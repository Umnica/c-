/*подставить среднее арфиметическое элементов, расположенных над побочной диагональю*/
#include <iostream>
using namespace std;

template <class T>
T average(T** a, int m, int n){
    T sum = 0;
    int count = 0;
    for (int i = 0; i < m; i++) {
        for (int j = i; j < n - 1; j++)
            if (a[i][j] != 0) {
                sum += a[i][j];
                count++;
            }
    }
    if (count == 0)
        return 0;
    cout << sum / count;
}
int main(){
    int m = 3, n = 3;
    int** a = new int*[m];
    for (int i = 0; i < m; i++)
        a[i] = new int[n];
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            a[i][j] = i+2;
        }
    }
    for (int i = 0; i < m; i++, cout << endl) {
        for (int j = 0; j < n; j++) {
            cout << a[i][j] << "\t";
        }
    }
    average(a, m, n);
    for (int i = 0; i < m; i++)
        delete[] a[i];
    return 0;
}
