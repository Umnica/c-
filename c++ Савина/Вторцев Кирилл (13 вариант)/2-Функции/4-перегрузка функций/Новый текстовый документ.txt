#include <iostream>
using namespace std;

float f(float x) {
    float y;
    if (abs(x) == 1)
        y = 1;
    else if (x < 0)
        y = -1 * x / (1 + x);
    else 
        y = 1 / (1 + x);
    return y;
}
void f(float x, float& y) {
    if (abs(x) == 1)
        y = 1;
    else if (x < 0)
        y = -1 * x / (1 + x);
    else
        y = 1 / (1 + x);
    cout << y << endl;
}
int main(){
    float a = 1.0;
    float y = f(a);
    cout << y << endl;
    f(a, y);
    cout << y << endl;
}
